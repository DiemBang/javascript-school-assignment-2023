const cardChecked = document.querySelector('#card'); const invoiceChecked = document.querySelector('#invoice'); //tick checkbox to unhide card payment details fields cardChecked.addEventListener('click', unhideCardFields); function unhideCardFields() { let checkBox = document.getElementById('card'); if (checkBox.checked === true) { document.getElementById('cardPayment').classList.remove('hidden'); } else { document.getElementById('cardPayment').classList.add('hidden'); } } //tick checkbox to unhide invoice fields invoiceChecked.addEventListener('click', unhideInvoiceFields); function unhideInvoiceFields() { let checkBox = document.getElementById('invoice'); if (checkBox.checked === true) { document.getElementById('invoicePayment').classList.remove('hidden'); } else { document.getElementById('invoicePayment').classList.add('hidden'); } }
